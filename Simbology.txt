#suports = Matrix that holds the nondisplaceable nodes. [Support Index , Acting Node , X-Freedom , Y-Freedom] 1 - Free to Displace ; 0 - Not Displaceable
#loadings = Matrix that holds all the loadings applied. [Force Index , Acting Node , X-Force , Y-Force]
